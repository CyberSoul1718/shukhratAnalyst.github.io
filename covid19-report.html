Step 3: Data Quality & Cleaning Operations

SQL 
-- Check for duplicates
SELECT code, continent, dates, country, populations, COUNT(*) AS Duplicates
FROM COVID19_Deaths
GROUP BY code, continent, dates, country, populations
HAVING COUNT(*) > 1;

-- Check for null continent values
SELECT *
FROM COVID19_Deaths
WHERE continent IS NULL;

-- Remove records with null population or continent
DELETE FROM COVID19_Deaths
WHERE populations IS NULL OR continent IS NULL;

-- Replace null values in numeric columns with 0
DECLARE @sql NVARCHAR(MAX) = '';
SELECT @sql = STRING_AGG(
    'UPDATE COVID19_Deaths SET ' + name + ' = COALESCE(' + name + ', 0);',
    ' '
)
FROM sys.columns 
WHERE object_id = OBJECT_ID('COVID19_Deaths')
AND system_type_id IN (56, 127, 62, 108, 106, 59);

EXEC sp_executesql @sql;

-- Convert date column from nvarchar to DATE data type
UPDATE COVID19_Deaths 
SET dates = TRY_CAST(dates AS DATE)
WHERE TRY_CAST(dates AS DATE) IS NOT NULL;

ALTER TABLE COVID19_Deaths
ALTER COLUMN dates date;

Purpose: Validate data integrity by checking for duplicates, handling null values, and converting data types to appropriate formats for analysis.

Step 4: Key Analysis Queries

SQL 
-- Mortality Rate by Country
-- Shows the percentage of cases that resulted in death
SELECT 
    country, 
    dates, 
    total_cases, 
    total_deaths,
    (total_deaths/total_cases)*100 AS mortality_rate
FROM dbo.COVID19_Deaths
WHERE total_cases > 0 
    AND continent IS NOT NULL
ORDER BY country, dates;

-- Infection Rate vs Population
-- Shows what percentage of the population was infected
SELECT 
    country, 
    dates, 
    total_cases, 
    populations,
    (total_cases/populations)*100 AS infection_rate
FROM dbo.COVID19_Deaths
WHERE continent IS NOT NULL  
ORDER BY country, dates;

-- Countries with Highest Infection Percentage
SELECT 
    country, 
    populations, 
    MAX(total_cases) AS HighestInfectionCount, 
    MAX(total_cases/populations)*100 AS InfectedPopulationPercentage 
FROM dbo.COVID19_Deaths
WHERE continent IS NOT NULL
GROUP BY country, populations
ORDER BY InfectedPopulationPercentage DESC;

-- Countries with Highest Death Counts
SELECT	
    country, 
    MAX(CAST(total_deaths AS INT)) AS Total_Death_Count 
FROM dbo.COVID19_Deaths
WHERE continent IS NOT NULL
GROUP BY country
ORDER BY Total_Death_Count DESC;

Purpose: Calculate key pandemic metrics including mortality rates, infection rates, and geographical comparisons to identify hardest-hit regions.
