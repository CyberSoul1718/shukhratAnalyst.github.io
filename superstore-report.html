Step 2: Data Cleaning & Preparation

Python 
import pandas as pd
from datetime import datetime

# Handle missing values
df.fillna(method='ffill', inplace=True)

# Convert date columns to datetime
df['Order Date'] = pd.to_datetime(df['Order Date'])
df['Ship Date'] = pd.to_datetime(df['Ship Date'])

# Calculate delivery days
df['Delivery_Days'] = (df['Ship Date'] - df['Order Date']).dt.days

# Create additional features
df['Year'] = df['Order Date'].dt.year
df['Month'] = df['Order Date'].dt.month
df['Quarter'] = df['Order Date'].dt.quarter
df['Day_of_Week'] = df['Order Date'].dt.day_name()

# Remove duplicates
df = df.drop_duplicates(subset=['Order ID'])

# Validate numeric fields
df['Sales'] = pd.to_numeric(df['Sales'], errors='coerce')
df['Profit'] = pd.to_numeric(df['Profit'], errors='coerce')
df['Quantity'] = pd.to_numeric(df['Quantity'], errors='coerce')

print("Data cleaning completed successfully!")
print(f"Final dataset shape: {df.shape}")

Purpose: Clean the dataset by handling missing values, converting data types, creating derived features, and removing duplicates to ensure data quality for analysis.

Step 3: Exploratory Data Analysis (EDA)

Python 
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Sales by Category
category_sales = df.groupby('Category')['Sales'].sum().sort_values(ascending=False)
print("Sales by Category:")
print(category_sales)

# Profit by Region
region_profit = df.groupby('Region')['Profit'].sum().sort_values(ascending=False)
print("\nProfit by Region:")
print(region_profit)

# Top 10 Products by Sales
top_products = df.groupby('Product Name')['Sales'].sum().nlargest(10)
print("\nTop 10 Products by Sales:")
print(top_products)

# Customer Segment Analysis
segment_analysis = df.groupby('Segment').agg({
    'Sales': 'sum',
    'Profit': 'sum',
    'Quantity': 'sum',
    'Order ID': 'count'
}).rename(columns={'Order ID': 'Order Count'})
print("\nCustomer Segment Analysis:")
print(segment_analysis)

# Monthly Sales Trend
monthly_sales = df.groupby(df['Order Date'].dt.to_period('M'))['Sales'].sum()
print("\nMonthly Sales Trend:")
print(monthly_sales)

Purpose: Perform comprehensive exploratory analysis to identify sales patterns, top performers, regional variations, and customer segments for strategic insights.

Step 4: Data Visualization & Insights

Python 
import matplotlib.pyplot as plt
import seaborn as sns

# Create visualization dashboard
fig, axes = plt.subplots(2, 2, figsize=(14, 10))

# 1. Sales by Category
df.groupby('Category')['Sales'].sum().plot(kind='bar', ax=axes[0, 0], color='#2ecc71')
axes[0, 0].set_title('Sales by Category', fontsize=12, fontweight='bold')
axes[0, 0].set_ylabel('Sales ($)')

# 2. Profit by Region
df.groupby('Region')['Profit'].sum().plot(kind='bar', ax=axes[0, 1], color='#3498db')
axes[0, 1].set_title('Profit by Region', fontsize=12, fontweight='bold')
axes[0, 1].set_ylabel('Profit ($)')

# 3. Sales vs Profit Scatter
axes[1, 0].scatter(df['Sales'], df['Profit'], alpha=0.5, color='#e74c3c')
axes[1, 0].set_title('Sales vs Profit Correlation', fontsize=12, fontweight='bold')
axes[1, 0].set_xlabel('Sales ($)')
axes[1, 0].set_ylabel('Profit ($)')

# 4. Monthly Sales Trend
df.set_index('Order Date')['Sales'].resample('M').sum().plot(ax=axes[1, 1], color='#f39c12')
axes[1, 1].set_title('Monthly Sales Trend', fontsize=12, fontweight='bold')
axes[1, 1].set_ylabel('Sales ($)')

plt.tight_layout()
plt.savefig('sales_analysis.png', dpi=300, bbox_inches='tight')
plt.show()

print("Visualizations created and saved!")

Purpose: Create comprehensive visualizations including category analysis, regional performance, sales-profit correlation, and temporal trends for presentation-ready insights.

Data Pipeline to Tableau

Python 
import pandas as pd

# Data aggregation for dashboard
dashboard_data = df.groupby(['Region', 'Category', 'Segment', 'Order Date']).agg({
    'Sales': 'sum',
    'Profit': 'sum',
    'Quantity': 'sum',
    'Order ID': 'count'
}).reset_index()

# Calculate KPIs
dashboard_data['Profit_Margin'] = (dashboard_data['Profit'] / dashboard_data['Sales'] * 100).round(2)
dashboard_data['Avg_Order_Value'] = (dashboard_data['Sales'] / dashboard_data['Order ID']).round(2)

# Export to Tableau-compatible format
dashboard_data.to_csv('super_store_dashboard_data.csv', index=False)

# Summary statistics for dashboard
print("Dashboard Data Summary:")
print(f"Total Records: {len(dashboard_data)}")
print(f"Date Range: {dashboard_data['Order Date'].min()} to {dashboard_data['Order Date'].max()}")
print(f"Average Profit Margin: {dashboard_data['Profit_Margin'].mean():.2f}%")
print("Data exported successfully!")

Purpose: Prepare and aggregate data in the optimal format for Tableau visualization, ensuring fast query performance and meaningful dashboard insights.
